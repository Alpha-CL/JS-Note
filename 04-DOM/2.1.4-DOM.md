#

## Insert

#### innerHTML

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * 设置或返回一个元素的 HTML 内容( 文字属性 )
 * 
 */

innerHTML: string;


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

#### innerText

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


// textContent 返回的文本内容的所有元素，而返回的 innerText所有元素的内容

/**
 * 设置或返回指定节点的文本内容( 及其所有后代 )
 * 
 * 输入覆盖: 设置的innerText属性，子节点将被删除 并包含指定字符串的单个Text节点 被替换
 * 
 * 火狐不兼容
 */

innerText: string;


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

#### textContent

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


// textContent 返回的文本内容的所有元素，而返回的 innerText所有元素的内容

/**
 * 设置或返回指定节点的文本内容( 及其所有后代 )
 * 
 * 输入覆盖: 设置的 textContent 属性，子节点将被删除 并包含指定字符串的单个Text节点 被替换
 * 
 * 老版本 IE 不兼容
 */
 
textContent: string | null;


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```


## Set

#### setAttribute

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * 为指定元素添加属性和值
 * 
 * 如果指定的属性已经存在，则值 只 设置 / 改变
 */

setAttribute(qualifiedName: string, value: string): void;


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

#### getAttribute

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * 返回指定元素的属性的值
 * 
 */

getAttribute(qualifiedName: string): string | null;


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```


## Example

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


// 根据标签名添加属性

<div></div>
<span></span>
<strong></strong>

<script>


    var allObj = document.getElementsByTagName('*');

    for (var i = 0; i < allObj.length; i++) {

        allObj[i].setAttribute('thisname', allObj[i].nodeName);
    }


</script>


//-------------------------------------------------------------------------------------------------------------------//


// 利用 JS 生成下面的 HTML 代码

<div class="example">
    <p class="slogan">Hello World</p>
</div>


<script>

    var dv = document.createElement('div'),
        pobj = document.createElement('p');
    
    dv.className = "example";
    pobj.className = "slogan";
    
    pobj.innerText = "Hello World";
    
    dv.appendChild(pobj);
    document.body.appendChild(dv);
    
</script>
```
