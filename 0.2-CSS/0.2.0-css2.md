# CSS 2.0

#### set css

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * 内部样式表
 * 
 * 
 */

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    
    <style>
    
        demo {
            
            /* css code block */
        }
    
    </style>
    
</head>
<body>

</body>
</html>


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


/**
 * 内联样式表
 * 
 * 
 */

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    
</head>
<body>

    <div style=" /* css code block */ ">hello world</div>

</body>
</html>


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


/**
 * 外部样式表
 * 
 * 
 */

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    
    
    <link rel="stylesheet" href="css/demo.css">                   // 引入外部 css 文件
    ...                                                           // 单个 html 文件可同时引入多个 css 文件
    
</head>
<body>

</body>
</html>


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
