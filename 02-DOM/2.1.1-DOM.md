#### Get Elements Methods

``` javascript
////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * elementId: 指定ID值的字符串，不区分大小写
 * 
 * IE8 以下:  ID 不区分大小写，可以获取指定( ID | NAME )
 * 返回具有指定 ID 或 NAME 属性值 的第一个对象的引用
 */
 
/**
 * ID 跟后端代码拼合的时候会 被替换掉
 * ID 一般作为顶级框架的名称
 * 选择器都用 class 即可
 */

getElementById(elementId: string): HTMLElement | null;                          // ID selector


//--------------------------------------------------------------------------------------------------------------------//


/**
 * 根据指定的元素名称检索对象的集合
 * objectArray: 类数组
 * 
 * IE4 都能用，兼容性好 主流用法 √
 */

getElementsByTagName(qualifiedName: string): HTMLCollectionOf<Element>;         // tag selector


// invoking

var tagObjs = getElementsByTagName('tagName');                      // [tagName#only]: 只有一个时，也是类数组

tagObjs[index: string];


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - //


<div class="wrap">
    <div class="con">
        <div class="side"></div>
    </div>
</div>

<script>

    var divObj = document.getElementsByTagName("div");                          // 获取 div 对象集合类数组
    
    divObj[0];                                                                  // 调用第一个 div 元素

</script>



//--------------------------------------------------------------------------------------------------------------------//


/**
 * 根据 NAME 或 ID 属性的值获取对象的集合
 * 
 * 注: 理论上只有部分标签的 name 可生效( 表单, 表单元素, img, iframe )
 */

getElementsByName(elementName: string): NodeListOf<HTMLElement>;                // 


//--------------------------------------------------------------------------------------------------------------------//


/**
 * 
 * 
 * IE8 和 IE8 以下的版本中不支持
 */

getElementsByClassName(classNames: string): HTMLCollectionOf<Element>;          // 


//--------------------------------------------------------------------------------------------------------------------//


// 静态选择，选取后不能再修改

/**
 * CSS 单个元素选择器
 * 
 * IE7 和 IE7 以下的版本中没有
 * 
 * return: 单个元素
 */


querySelector<E extends Element = Element>(selectors: string): E | null;


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - //


/**
 * CSS 多个个元素选择器
 * 
 * IE7 和 IE7 以下的版本中没有
 * 
 * return: 类数组 ｜ [element#only] 一个也是类数组类型
 */

querySelectorAll<E extends Element = Element>(selectors: string): NodeListOf<E>;


////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

#### Node Properties


``` javascript
////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


// 节点类型对应数组

elementNode             // 1

attributeNode           // 2

textNode                // 3

comment                 // 8

document                // 9

DocumentFragment        // 11

/**
 * 根据节点类型返回类型对应的数字
 * 
 * 只读
 */


readonly nodeType: number;


//--------------------------------------------------------------------------------------------------------------------//


/**
 * 元素名称，以大写形式表示
 * 
 * 只读
 */


readonly nodeName: string;


//--------------------------------------------------------------------------------------------------------------------//


/**
 * text 节点 或 comment 节点的文本内容
 * 
 * 可读写
 */

nodeValue: string | null;


//--------------------------------------------------------------------------------------------------------------------//


/**
 * 节点的属性集合
 * 
 * 
 */

readonly attributes: NamedNodeMap;


//--------------------------------------------------------------------------------------------------------------------//


/**
 * 是否有字节点
 * 
 * 
 */


hasChildNodes(): boolean;


//--------------------------------------------------------------------------------------------------------------------//






// 模拟系统 Children();


function getElementChild(node) {

    var temp = {

            length : 0,
            push : Array.prototype.push,
            splice : Array.prototype.splice
        },
        child = node.childNodes,
        len = child.length;
    
    for (var i = 0; i < len; i++) {

        if (child[i].nodeType === 1) {

            temp.push(child[i]);

        }
    }
    
    return temp;
}
```
